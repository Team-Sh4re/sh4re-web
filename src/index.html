def get_data():
a1, b1 = map(int, input().split())
a2, b2 = map(int, input().split())
return a1, b1, a2, b2

def get_line(a1, b1, a2, b2):
if a1 == a2:
  rst = f'x = {a1}'
else:
  m = (b2 - b1) / (a2 - a1)
  y_i = b1 - m * a1
  rst = f'y = {m}x + ({y_i})'
return rst

x1, y1, x2, y2 = get_data()
line = get_line(x1, y1, x2, y2)
print(line)

import math
lst = list(range(1, 11))

# 평균
mean = sum(lst)/len(lst)
print(f"평균:{mean}")

# 분산
vsum = 0
for x in lst:
  vsum += (x-mean)**2
var = vsum / len(lst)
print(f"분산: {var}")

# 표준편차
std = math.sqrt(var)
print(f"표준편차: {std}")

# 최대공약수, 최소공배수
gcd = math.gcd(*lst)
lcm = math.lcm(*lst)
print(f"최대공약수: {gcd} 최소공배수: {lcm}")
